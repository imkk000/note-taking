# Code
```go
func divideArray(nums []int) bool {
    ds := len(nums)/2
    v := make([]int, 501)
    for _, n := range nums {
        v[n]++
    }
    for _, n := range v {
        if n % 2 != 0 || ds < 0 {
            return false
        }
        ds -= n/2
    }
    return true
}
```
